# 冒泡排序

原理：

从第一个元素开始，往后比较，遇到比自己小的元素就交换位置

特点：

交换的次数最多，所以它的性能是最差的

function bubbleSort(arr){
    let len = arr.length;
    // 外层循环
    console.time("timer");
    for (let i = 0; i < len - 1; i++) {
        // 内层循环
        for (let j = 0; j < len - i; j++) {
        // 判断
            if (arr[j] > arr[j + 1]) {
                // ES6 的交换数值
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
            }
        }
    }
    console.timeEnd("timer");
    return arr;
}
let arr=[2, 3, 6, 4, 2, 1, 90, 100, 20, 5];
console.log(bubbleSort(arr)); 
// 花费 timer: 0.092041015625ms
// [1, 2, 2, 3, 4, 5, 6, 20, 90, 100]

//插入排序
//假定当前元素之前的元素已经排好序，先把自己的位置空出来，
//然后前面比自己大的元素依次向后移，直到空出一个"坑"，
//然后把目标元素插入"坑"中
function insertSort(arr){
   
    console.time("timer")
     // i = 1 是因为从第二个元素开始，因为要留出一个坑
    let len = arr.length
    for (let i = 1; i < len; i++) {
        let key = arr[i]
        //已经排序好的数据的尾的索引
        let j = i - 1
        
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j]
            j--
        }
        arr[j + 1] = key
    }
    console.timeEnd("timer")
    return arr
}
let arr = [2, 3, 6, 4, 2, 1, 90, 100, 20, 5]
console.log(insertSort(arr))
// timer: 0.039794921875ms
// [1, 2, 2, 3, 4, 5, 6, 20, 90, 100]
