    # 我的解决办法

    1. 将一维数据转成多维树状数据
    2. 从多维树状数据中寻找，符合模糊查询条件的数据
    3. 将多维转成一维数据

    # 开始实施

    1.将一维数据转成多维树状数据

     /**1.将一维数据转成多维树状数据*/
    lookFor: function (data) {
        //这里采用的是一个递归的方式
        function toTree(list,parId){
           let len = list.length
           function loop(parId){
                let res = []
                //循环一层
                for(let i = 0; i < len; i++){
                    let item = list[i]  
                    if (item.pid === parId) {
                        var children =  loop(item.id)
                        if (children.length != 0) {
                            item.children = children
                        }
                            res.push(item)
                        }
                    }
                return res
            }
            return loop(parId)
        }
        //都遍历完才知道谁有父级
        return toTree(data, '#')
    },

dataFilter: function (item, shopSearch) {
    var _this = this
    if (item.name.indexOf(shopSearch) > -1) return true;
       if (item.hasOwnProperty("children")) {
       item.children = item.children.filter(function (child) {
            if (!child.hasOwnProperty("children")) {
                return child.name.indexOf(shopSearch) > -1;
            } else {
                return _this.dataFilter(child);
            }
        })
        if (item.children.length > 0) {
            return true;
        }
        } else {
            return child.name.indexOf(shopSearch) > -1;

}

    1. 将多维转成一维数据
    // 多维树形数据转成一维数据
    DimensionReduction: function (source){
        var res = []
        function fn(source) {
            source.forEach(function(el){
                res.push(el)
                el.children && el.children.length>0 ? fn(el.children) : ""        // 子级递归
            })
        }
        fn(source)
        res.forEach(function (item) {
            item.children = null
        })
        return res
    }
